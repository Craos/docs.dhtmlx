StartFormForm APIProperties

title

specifies the title of the control group

string title;

Example

var form = new dhx.Form("form_container", {
    css: "dhx_widget--bordered",
    title: "Form",  // title for rows    rows: [
        {
            title: "Controls",  // title for cols            css: "dhx_layout-cell--bordered",
            cols: [{
                type: "checkbox",
                label: "start",
                name: "align",
                gravity: false,
                labelInline: true,
                value: "start",
                checked: true
            },
            {
                type: "checkbox",
                label: "center",
                gravity: false,
                name: "align",
                labelInline: true,
                value: "center"
            }]
        }
    ]
});
